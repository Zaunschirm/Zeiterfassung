Wichtige Befehle für Zeiterfassung:

PW zurücksetzen ZS: in SQL:

-- ZS aktivieren, nicht deaktiviert, Rolle admin, PIN auf "0000"
create extension if not exists pgcrypto;

update public.employees
set role = 'admin',
    active = true,
    disabled = false,
    pin_hash = crypt('0000', gen_salt('bf')),
    updated_at = now()
where lower(code) = 'zs';
